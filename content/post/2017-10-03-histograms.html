library(Lahman)
library(sqldf)
library(ggplot2)

Distribution of the weight of baseball players

query<-"SELECT weight
FROM Master"

result<-sqldf(query)
ggplot()+
  geom_histogram(data=result,aes(x=weight),color="blue",fill="white",bins=50)+
  ggtitle("Body-weight Distribution for Baseball Players")
