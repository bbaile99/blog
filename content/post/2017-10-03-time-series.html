library(Lahman)

library(ggplot2)

query<-"SELECT yearID,HR
FROM Batting
WHERE playerID='ruthba01'"

result<-sqldf(query)
ggplot()+
  geom_point(data=result,aes(x=yearID,y=HR))+
  geom_line(data=result,aes(x=yearID,y=HR))+
  ggtitle("Ruth's Homeruns Totals Throughout the Years")+
  xlab("year")+
  ylab("homeruns")
